```csharp
                case "social": social = Mathf.Clamp(social + kv.Value, 0, 100); break;
                case "health": health = Mathf.Clamp(health + kv.Value, 0, 100); break;
            }
        }
    }

        choiceHistoryList.Add(new StringPair { key = eventId, value = choiceId });
```

### 2.1 EventEngine.cs - 事件引擎

```csharp
public class EventEngine : MonoBehaviour
{
    private List<GameEvent> allEvents = new List<GameEvent>();
    private List<string> completedEventIds = new List<string>();
    private PlayerState playerState;
    private System.Random rng;

    public void SetPlayerState(PlayerState state)
    {
        playerState = state;
        if (state != null)
        {
            completedEventIds = new List<string>(state.completedEvents);
            rng = new System.Random();
        }
    }

    public IEnumerator LoadEventsFromStreamingAssets(string fileName)
    {
        string path = System.IO.Path.Combine(Application.streamingAssetsPath, "Events", $"{fileName}.json");
        UnityWebRequest request = UnityWebRequest.Get(path);
        yield return request.SendWebRequest();

        if (request.result == UnityWebRequest.Result.Success)
        {
            string json = request.downloadHandler.text;
            var wrapper = JsonHelper.GetJsonList<GameEvent>(json);
            allEvents = wrapper ?? new List<GameEvent>();
            Debug.Log($"[EventEngine] 加载事件: {allEvents.Count}个");
        }
        else
        {
            Debug.LogError($"[EventEngine] 加载失败: {request.error}");
        }
    }

    public GameEvent GetNextEvent()
    {
        if (allEvents.Count == 0) return null;

        var candidate = allEvents
            .Where(e => !completedEventIds.Contains(e.id))
            .Where(e => MeetsCondition(e))
            .ToList();

        if (candidate.Count == 0) return null;

        return candidate[rng.Next(candidate.Count)];
    }

    public bool MeetsCondition(GameEvent e)
    {
        if (e.triggerCondition == null) return true;
        if (playerState == null) return true;

        if (e.triggerCondition.minGrade > 0 && playerState.grade < e.triggerCondition.minGrade)
            return false;
        if (e.triggerCondition.minIntellect > 0 && playerState.intellect < e.triggerCondition.minIntellect)
            return false;
        if (!string.IsNullOrEmpty(e.triggerCondition.requiredEvent) &&
            !completedEventIds.Contains(e.triggerCondition.requiredEvent))
            return false;

        return true;
    }

    public EventOutcome ResolveEvent(string eventId, string choiceId)
    {
        var evt = allEvents.FirstOrDefault(e => e.id == eventId);
        if (evt == null) return null;

        completedEventIds.Add(eventId);
        if (playerState != null)
        {
            if (!playerState.completedEvents.Contains(eventId))
                playerState.completedEvents.Add(eventId);
            playerState.RecordChoice(eventId, choiceId);
        }

        var outcomes = evt.GetOutcomeDict();
        if (outcomes.TryGetValue(choiceId, out var outcome))
        {
            ApplyOutcome(outcome);
            return outcome;
        }

        return null;
    }

    private void ApplyOutcome(EventOutcome outcome)
    {
        if (playerState == null || outcome == null) return;

        var changes = outcome.GetStatChanges();
        playerState.ApplyChanges(changes);

        if (outcome.unlockEvents != null)
        {
            foreach (var unlockId in outcome.unlockEvents)
            {
                Debug.Log($"[EventEngine] 解锁事件: {unlockId}");
            }
        }
    }

    public GameEvent GetEventById(string eventId)
    {
        return allEvents.FirstOrDefault(e => e.id == eventId);
    }

    public List<GameEvent> GetEventsByStage(string stage)
    {
        return allEvents.Where(e => e.stage == stage).ToList();
    }

    public float GetEventProbability(GameEvent evt)
    {
        if (playerState == null) return 0.5f;

        float baseProb = 0.5f;
        float intellectFactor = playerState.intellect / 100f * 0.3f;
        float mentalFactor = playerState.mental / 100f * 0.2f;

        return Mathf.Clamp01(baseProb + intellectFactor + mentalFactor);
    }
}
```

```csharp
public class TalentEngine : MonoBehaviour
{
    public static TalentEngine Instance { get; private set; }
    private List<Talent> allTalents = new List<Talent>();
    private Talent currentTalent;
    private System.Random rng = new System.Random();

    void Awake()
    {
        if (Instance == null) { Instance = this; DontDestroyOnLoad(gameObject); }
        else { Destroy(gameObject); }
    }

    public void LoadTalents(List<Talent> talents)
    {
        allTalents = talents ?? new List<Talent>();
    }

    public List<Talent> DrawTalents(int count = 3)
    {
        if (allTalents.Count == 0) return new List<Talent>();
        count = Mathf.Min(count, allTalents.Count);

        var candidates = allTalents
            .OrderBy(t => -t.probability)
            .ThenBy(_ => rng.Next())
            .Take(count * 3)
            .ToList();

        return candidates.OrderBy(_ => rng.Next()).Take(count).ToList();
    }

    public void SelectTalent(string talentId)
    {
        currentTalent = allTalents.FirstOrDefault(t => t.id == talentId);
        Debug.Log($"[TalentEngine] 选择天赋: {currentTalent?.name ?? talentId}");
    }

    public Talent GetCurrentTalent() => currentTalent;

    public float GetTalentBonus(string statName)
    {
        if (currentTalent?.effects == null) return 1f;

        foreach (var effect in currentTalent.effects)
        {
            if (effect.type == "stat_multiply" && effect.statKey == statName)
                return 1f + effect.value / 100f;
        }
        return 1f;
    }

    public void ApplyTalentEffect(PlayerState state)
    {
        if (currentTalent?.effects == null || state == null) return;

        foreach (var effect in currentTalent.effects)
        {
            switch (effect.type)
            {
                case "stat_add":
                    state.AddStat(effect.statKey, effect.value);
                    break;
                case "gaokao_bonus":
                    Debug.Log($"[TalentEngine] 高考加成: {effect.value}");
                    break;
            }
        }
    }

    public int GetGaokaoBonus()
    {
        if (currentTalent?.effects == null) return 0;

        foreach (var effect in currentTalent.effects)
        {
            if (effect.type == "gaokao_bonus")
                return effect.value;
        }
        return 0;
    }
}
```

```csharp
public class ScoreEngine : MonoBehaviour
{
    public static ScoreEngine Instance { get; private set; }
    private System.Random rng = new System.Random();

    void Awake()
    {
        if (Instance == null) { Instance = this; DontDestroyOnLoad(gameObject); }
        else { Destroy(gameObject); }
    }

    public int CalculateScore(PlayerState state, string talentId = "")
    {
        float intellectScore = GetSubjectScore("语文", state.intellect)
                             + GetSubjectScore("数学", state.intellect)
                             + GetSubjectScore("英语", state.intellect);

        float scienceScore = GetSubjectScore("物理", state.intellect)
                          + GetSubjectScore("化学", state.intellect)
                          + GetSubjectScore("生物", state.intellect);

        float liberalScore = GetSubjectScore("政治", state.intellect)
                           + GetSubjectScore("历史", state.intellect)
                           + GetSubjectScore("地理", state.intellect);

        bool isScience = state.grade >= 3;
        float baseScore = isScience ? scienceScore : liberalScore;
        float totalBase = intellectScore + baseScore;

        float mentalBonus = (state.mental - 50) * 1.5f;
        float mentalVariation = rng.Next(-15, 16);

        float talentBonus = GetTalentBonus(talentId);

        float finalScore = totalBase + mentalBonus + mentalVariation + talentBonus;
        return Mathf.RoundToInt(Mathf.Clamp(finalScore, 0, 750));
    }

    public int GetSubjectScore(string subject, int baseStat)
    {
        float baseScore = baseStat * 0.7f + rng.Next(-5, 6);
        return Mathf.RoundToInt(Mathf.Clamp(baseScore, 0, 150));
    }

    private float GetTalentBonus(string talentId)
    {
        if (string.IsNullOrEmpty(talentId)) return 0;

        var bonusDict = new Dictionary<string, float>
        {
            { "T01", 25f },
            { "T02", 30f },
            { "T03", 15f },
            { "T06", 20f },
            { "T07", 15f },
        };

        return bonusDict.TryGetValue(talentId, out float bonus) ? bonus : 0f;
    }

    public Dictionary<string, int> GetDetailedScores(PlayerState state, string talentId)
    {
        var scores = new Dictionary<string, int>
        {
            { "语文", GetSubjectScore("语文", state.intellect) },
            { "数学", GetSubjectScore("数学", state.intellect) },
            { "英语", GetSubjectScore("英语", state.intellect) },
        };

        bool isScience = state.grade >= 3;
        if (isScience)
        {
            scores["物理"] = GetSubjectScore("物理", state.intellect);
            scores["化学"] = GetSubjectScore("化学", state.intellect);
            scores["生物"] = GetSubjectScore("生物", state.intellect);
        }
        else
        {
            scores["政治"] = GetSubjectScore("政治", state.intellect);
            scores["历史"] = GetSubjectScore("历史", state.intellect);
            scores["地理"] = GetSubjectScore("地理", state.intellect);
        }

        return scores;
    }
}
```

```csharp
public class PersonalityEngine : MonoBehaviour
{
    public static PersonalityEngine Instance { get; private set; }
    private List<Personality> allPersonalities = new List<Personality>();

    void Awake()
    {
        if (Instance == null) { Instance = this; DontDestroyOnLoad(gameObject); }
        else { Destroy(gameObject); }
    }

    public void LoadPersonalities(List<Personality> personalities)
    {
        allPersonalities = personalities ?? new List<Personality>();
    }

    public PersonalityResult GetPersonalityScores(PlayerState state)
    {
        if (state == null) return null;

        int intellect = state.intellect;
        int mental = state.mental;
        int social = state.social;
        int health = state.health;

        int maxStat = Mathf.Max(intellect, mental, social, health);
        PersonalityType dominant = PersonalityType.Balanced;

        if (maxStat == intellect) dominant = PersonalityType.Academic;
        else if (maxStat == mental) dominant = PersonalityType.Perfectionist;
        else if (maxStat == social) dominant = PersonalityType.Social;
        else if (maxStat == health) dominant = PersonalityType.Athletic;

        int rational = Mathf.RoundToInt((intellect * 0.6f + mental * 0.3f + social * 0.1f));
        int impulsive = Mathf.RoundToInt((mental * 0.5f + social * 0.3f + health * 0.2f));
        int adventurous = Mathf.RoundToInt((social * 0.4f + health * 0.4f + mental * 0.2f));

        var personality = allPersonalities.FirstOrDefault(p => p.id == dominant.ToString().ToLower())
                      ?? allPersonalities.FirstOrDefault();

        return new PersonalityResult
        {
            personality = personality,
            rational = Mathf.Clamp(rational, 0, 100),
            impulsive = Mathf.Clamp(impulsive, 0, 100),
            adventurous = Mathf.Clamp(adventurous, 0, 100)
        };
    }

    public PersonalityType DeterminePersonality(PlayerState state)
    {
        if (state == null) return PersonalityType.Balanced;

        int intellect = state.intellect;
        int mental = state.mental;
        int social = state.social;
        int health = state.health;

        int maxStat = Mathf.Max(intellect, mental, social, health);

        if (maxStat == intellect && intellect - mental >= 10) return PersonalityType.Academic;
        if (maxStat == mental && mental - intellect >= 10) return PersonalityType.Perfectionist;
        if (maxStat == social) return PersonalityType.Social;
        if (maxStat == health) return PersonalityType.Athletic;

        return PersonalityType.Balanced;
    }
}
```

```csharp
public class DejaVuEngine : MonoBehaviour
{
    public static DejaVuEngine Instance { get; private set; }
    private int maxLevel = 0;
    private List<PlaythroughMemory> pastMemories = new List<PlaythroughMemory>();
    private HashSet<string> rememberedChoices = new HashSet<string>();

    void Awake()
    {
        if (Instance == null) { Instance = this; DontDestroyOnLoad(gameObject); }
        else { Destroy(gameObject); }
    }

    public void SetMaxLevel(int level)
    {
        maxLevel = Mathf.Clamp(level, 0, 4);
        Debug.Log($"[DejaVuEngine] 既视感等级: {maxLevel}");
    }

    public int GetMaxDejaVuLevel() => maxLevel;

    public void LoadMemories(List<PlaythroughMemory> memories)
    {
        pastMemories = memories ?? new List<PlaythroughMemory>();
        rememberedChoices.Clear();

        foreach (var mem in pastMemories)
        {
            if (mem.finalStatsList != null)
            {
                foreach (var stat in mem.finalStatsList)
                {
                    rememberedChoices.Add($"{mem.talentId}_{stat.key}_{stat.value}");
                }
            }
        }
    }

    public DejaVuResult CheckDejaVu(string eventId, int level)
    {
        if (level <= 0) return null;

        var memory = pastMemories.LastOrDefault(m =>
            m.finalStatsList?.Any(s => true) == true);

        if (memory == null) return null;

        switch (level)
        {
            case 1:
                return new DejaVuResult { hint = "这个场景似乎有些熟悉..." };
            case 2:
                return new DejaVuResult
                {
                    hint = "你隐约记得某个选择会带来好的结果",
                    highlightChoiceId = "A"
                };
            case 3:
                return new DejaVuResult
                {
                    hint = $"上一次你选择了A，那次经历{nmemory.finalCollege ?? "很顺利"}",
                    highlightChoiceId = "A"

                }
            }
        }
```

```csharp
        {
            if (!data.alumniColleges.Contains(lastGenMemory.finalCollege))
                data.alumniColleges.Add(lastGenMemory.finalCollege);
        }

        Debug.Log($"[InheritEngine] 开始第{data.generation}代, 传承属性: {data.inheritedStatsList.Count}项");
    }
```

```csharp
    {
        mergeRules.Clear();
        if (keys != null && values != null)
        {
            for (int i = 0; i < Mathf.Min(keys.Length, values.Length); i++)
                mergeRules[keys[i]] = values[i];
        }
    }

    public string TryMerge(string itemA, string itemB)
    {
        string key1 = $"{itemA}+{itemB}";
        string key2 = $"{itemB}+{itemA}";

        if (mergeRules.TryGetValue(key1, out string result)) return result;
        if (mergeRules.TryGetValue(key2, out string result2)) return result2;

        return null;
    }

    public bool CheckMergeResult(string itemA, string itemB, string targetItem)
    {
        string merged = TryMerge(itemA, itemB);
        return merged == targetItem;
    }

    public int CalculateMergeScore(string itemA, string itemB, string target)
    {
        bool success = CheckMergeResult(itemA, itemB, target);
        return success ? 100 : 0;
    }
```

```csharp
    {
```

```csharp
        yield return request.SendWebRequest();

        if (request.result == UnityWebRequest.Result.Success)

            string json = request.downloadHandler.text;
            var list = JsonHelper.GetJsonList<T>(json);
            onLoaded?.Invoke(list ?? new List<T>());
        }
        else
        {
            Debug.LogError($"[DataLoader] 加载失败: {fileName} - {request.error}");
            onLoaded?.Invoke(new List<T>());
        }
    }
```

```csharp
            genderSelectUI.OnGenderSelected += OnGenderSelected;
            genderSelectUI.Show();
        }
        else
        {
            genderSelected = true;
            if (mainMenuPanel != null)
                mainMenuPanel.SetActive(true);
        }
    }

    private void OnGenderSelected(Gender gender)
    {
        genderSelected = true;
        if (genderSelectUI != null)
            genderSelectUI.OnGenderSelected -= OnGenderSelected;

        if (mainMenuPanel != null)
            mainMenuPanel.SetActive(true);

        Debug.Log($"[HomeUI] 性别已选择: {gender}");
    }

    void OnStartGame()
    {
        if (!genderSelected)
        {
            Debug.LogWarning("[HomeUI] 请先选择性别");
            return;
        }

        if (GameStateManager.Instance != null)
            GameStateManager.Instance.NewGame();

        Debug.Log("[HomeUI] 开始新游戏");
        UnityEngine.SceneManagement.SceneManager.LoadScene("HighSchool");
    }

    void OnContinueGame()
    {
        if (GameStateManager.Instance != null)
            GameStateManager.Instance.QuickLoad();
    }

    void OnSettings()
    {
        Debug.Log("[HomeUI] 打开设置");
    }
}
```

```csharp
public class HighSchoolUI : MonoBehaviour
{
    [Header("事件卡片")]
    [SerializeField] private GameObject eventCardPanel;
    [SerializeField] private Image eventIllustration;
    [SerializeField] private Image categoryIcon;
    [SerializeField] private Text eventNameText;
    [SerializeField] private Transform interactionArea;
    [SerializeField] private Button advanceTimeBtn;

    private PlayerState playerState;
    private GameEvent currentEvent;
    private EventEngine eventEngine;
    private DejaVuEngine dejaVuEngine;

    void Start()
    {
        eventEngine = FindObjectOfType<EventEngine>();
        dejaVuEngine = FindObjectOfType<DejaVuEngine>();

        playerState = GameStateManager.Instance?.GetPlayerState();
        if (playerState == null)
        {
            playerState = new PlayerState();
            GameStateManager.Instance?.SetPlayerState(playerState);
        }

        if (eventEngine != null)
        {
            eventEngine.SetPlayerState(playerState);
            StartCoroutine(LoadAndBegin());
        }

        advanceTimeBtn?.onClick.AddListener(OnAdvanceTime);
        if (resultPanel != null) resultPanel.SetActive(false);
        if (dejaVuBubble != null) dejaVuBubble.SetActive(false);

        BuildTimeline();
        RefreshTopBar();
    }

    IEnumerator LoadAndBegin()
    {
        yield return StartCoroutine(eventEngine.LoadEventsFromStreamingAssets("highschool"));
        ShowNextEvent();
    }

    void ShowNextEvent()
    {
        if (eventEngine == null) return;

        currentEvent = eventEngine.GetNextEvent();
        if (currentEvent == null)
        {
            Debug.Log("[HighSchoolUI] 没有更多事件");
            if (eventCardPanel != null) eventCardPanel.SetActive(false);
            return;
        }

        if (eventCardPanel != null) eventCardPanel.SetActive(true);
        if (eventNameText != null) eventNameText.text = currentEvent.name;

        LoadEventIllustration();
        ClearInteractionArea();
        BuildInteractionArea();
        CheckDejaVu();
    }

    void LoadEventIllustration()
    {
        if (eventIllustration == null || currentEvent == null) return;

        Gender gender = GameStateManager.Instance?.GetPlayerGender() ?? Gender.Male;
        string illustrationId = ResourceHelper.GetIllustrationIdForEvent(currentEvent.id, currentEvent.stage);
        var sprite = ResourceHelper.LoadEventIllustration(illustrationId, gender, currentEvent.stage);
        if (sprite != null) eventIllustration.sprite = sprite;
    }

    void BuildChoiceArea()
    {
        var choices = currentEvent.choiceContent?.choices;
        if (choices == null) return;

        foreach (var choice in choices)
        {
            var btnGo = Instantiate(choiceBtnPrefab, interactionArea);
            var btnText = btnGo.GetComponentInChildren<Text>();
            if (btnText != null)
                btnText.text = $"{choice.icon} {choice.text}";

            var btn = btnGo.GetComponent<Button>();
            string cid = choice.id;
            if (btn != null)
                btn.onClick.AddListener(() => OnChoiceSelected(cid));
        }
    }

    void OnChoiceSelected(string choiceId)
    {
        EventOutcome outcome = eventEngine.ResolveEvent(currentEvent.id, choiceId);
        if (outcome == null) return;

        RefreshTopBar();
        ShowResult(outcome);
    }

    void ShowResult(EventOutcome outcome)
    {
        if (resultPanel == null) return;
        resultPanel.SetActive(true);

        if (resultNarrativeText != null)
            resultNarrativeText.text = outcome.narrative;

        if (resultStatChangesText != null)
        {
            var changes = outcome.GetStatChanges();
            var parts = new List<string>();
            foreach (var kv in changes)
            {
                string sign = kv.Value >= 0 ? "+" : "";
                parts.Add($"{kv.Key}: {sign}{kv.Value}");
            }
            resultStatChangesText.text = string.Join("  ", parts);
        }

        if (resultContinueBtn != null)
        {
            resultContinueBtn.onClick.RemoveAllListeners();
            resultContinueBtn.onClick.AddListener(OnContinueAfterResult);
        }
    }

    void OnContinueAfterResult()
    {
        if (resultPanel != null) resultPanel.SetActive(false);
        if (dejaVuBubble != null) dejaVuBubble.SetActive(false);

        GameStateManager.Instance?.QuickSave();
        ShowNextEvent();
    }

    void OnAdvanceTime()
    {
        playerState.grade = Mathf.Min(playerState.grade + 1, 6);
        RefreshTopBar();
        RefreshTimeline();

        GameStateManager.Instance?.QuickSave();
        ShowNextEvent();
    }
}
```

## 6. GameManager - 总控制器

```csharp
public class GameManager : MonoBehaviour
{
    public static GameManager Instance { get; private set; }

    private PlayerState currentPlayer;
    private GamePhase currentPhase;
    private bool isGameRunning;
    private bool isPaused;

    private List<string> gameSceneNames = new List<string>
    {
        "Home", "Province", "HighSchool", "Event",
        "Gaokao", "Zhiyuan", "College", "Life", "Result"
    };

    public event System.Action<PlayerState> OnPlayerStateChanged;
    public event System.Action<GamePhase> OnPhaseChanged;
    public event System.Action OnGameStarted;
    public event System.Action OnGameEnded;

    public PlayerState CurrentPlayer => currentPlayer;
    public GamePhase CurrentPhase => currentPhase;
    public bool IsGameRunning => isGameRunning;
    public bool IsPaused => isPaused;

    void Awake()
    {
        if (Instance == null)
        {
            Instance = this;
            DontDestroyOnLoad(gameObject);
        }
        else
        {
            Destroy(gameObject);
            return;
        }

        Initialize();
    }

    private void Initialize()
    {
        currentPhase = GamePhase.Home;
        isGameRunning = false;
        isPaused = false;
        Debug.Log("[GameManager] 游戏管理器初始化完成");
    }

    void Start()
    {
        CheckForSavedGame();
    }

    private void CheckForSavedGame()
    {
        if (SaveManager.Instance != null && SaveManager.Instance.HasSaveData())
        {
            Debug.Log("[GameManager] 发现存档，可以继续游戏");
        }
        else
        {
            Debug.Log("[GameManager] 没有存档，创建新游戏");
        }
    }

    public void StartNewGame()
    {
        Gender savedGender = Gender.Male;
        if (SaveManager.Instance != null)
        {
            var existing = SaveManager.Instance.GetCurrentPlayer();
            if (existing != null) savedGender = existing.gender;
        }

        if (SaveManager.Instance != null)
        {
            SaveManager.Instance.CreateNewGame();
            currentPlayer = SaveManager.Instance.GetCurrentPlayer();
            if (currentPlayer != null) currentPlayer.gender = savedGender;
        }
        else
        {
            currentPlayer = new PlayerState();
        }

        isGameRunning = true;
        currentPhase = GamePhase.Home;

        LoadGameData();

        OnPlayerStateChanged?.Invoke(currentPlayer);
        OnPhaseChanged?.Invoke(currentPhase);
        OnGameStarted?.Invoke();

        Debug.Log("[GameManager] 开始新游戏");
        LoadScene("Home");
    }

    public void ContinueGame()
    {
        if (SaveManager.Instance != null && SaveManager.Instance.LoadGame())
        {
            currentPlayer = SaveManager.Instance.GetCurrentPlayer();
            isGameRunning = true;

            LoadGameData();

            OnPlayerStateChanged?.Invoke(currentPlayer);
            OnPhaseChanged?.Invoke(currentPlayer.currentPhase);

            Debug.Log("[GameManager] 继续游戏");
            LoadScene(currentPlayer.currentPhase.ToString());
        }
        else
        {
            Debug.LogWarning("[GameManager] 无法加载存档");
        }
    }

    private void LoadGameData()
    {
        if (DataLoader.Instance != null)
        {
            DataLoader.Instance.LoadAllData(() =>
            {
                Debug.Log("[GameManager] 游戏数据加载完成");
            });
        }
    }

    public void ChangePhase(GamePhase newPhase)
    {
        if (currentPhase == newPhase) return;

        currentPhase = newPhase;
        currentPlayer.currentPhase = newPhase;

        OnPhaseChanged?.Invoke(newPhase);
        Debug.Log($"[GameManager] 阶段切换: {newPhase}");

        LoadScene(newPhase.ToString());
    }

    public void LoadScene(string sceneName)
    {
        if (gameSceneNames.Contains(sceneName))
        {
            SceneManager.LoadScene(sceneName);
            Debug.Log($"[GameManager] 加载场景: {sceneName}");
        }
        else
        {
            Debug.LogWarning($"[GameManager] 场景不存在: {sceneName}");
        }
    }

    public void SaveGame(int slotIndex = 0)
    {
        if (currentPlayer == null)
        {
            Debug.LogWarning("[GameManager] 没有玩家数据可保存");
            return;
        }

        currentPlayer.lastPlayTime = System.DateTime.Now;

        if (SaveManager.Instance != null)
        {
            SaveManager.Instance.SetCurrentPlayer(currentPlayer);
            SaveManager.Instance.SaveGame(slotIndex);
        }
    }

    public void PauseGame()
    {
        isPaused = true;
        Time.timeScale = 0f;
        Debug.Log("[GameManager] 游戏暂停");
    }

    public void ResumeGame()
    {
        isPaused = false;
        Time.timeScale = 1f;
        Debug.Log("[GameManager] 游戏继续");
    }

    public void EndGame()
    {
        isGameRunning = false;
        SaveGame();
        OnGameEnded?.Invoke();
        Debug.Log("[GameManager] 游戏结束");
    }

    public void UpdatePlayerState(System.Action<PlayerState> updateAction)
    {
        if (currentPlayer == null) return;
        updateAction?.Invoke(currentPlayer);
        OnPlayerStateChanged?.Invoke(currentPlayer);
    }

    public void AddDays(int days)
    {
        if (currentPlayer == null) return;

        currentPlayer.studyDays += days;

        if (currentPlayer.studyDays >= 1095 && currentPhase == GamePhase.HighSchool)
        {
            ChangePhase(GamePhase.Gaokao);
        }

        OnPlayerStateChanged?.Invoke(currentPlayer);
    }

    public int CalculateGaokaoScore()
    {
        if (ScoreEngine.Instance != null)
        {
            string talentId = TalentEngine.Instance?.GetCurrentTalent()?.id ?? "";
            return ScoreEngine.Instance.CalculateScore(currentPlayer, talentId);
        }
        return 400;
    }

    public void StartNewGamePlus()
    {
        Gender savedGender = Gender.Male;
        var existing = SaveManager.Instance?.GetCurrentPlayer();
        if (existing != null) savedGender = existing.gender;

        SaveManager.Instance?.CreateNewGame();
        currentPlayer = SaveManager.Instance?.GetCurrentPlayer() ?? new PlayerState();
        currentPlayer.gender = savedGender;
        currentPlayer.hasPastLifeMemory = true;

        isGameRunning = true;
        ChangePhase(GamePhase.HighSchool);

        Debug.Log("[GameManager] 开始二周目");
    }
}
```
